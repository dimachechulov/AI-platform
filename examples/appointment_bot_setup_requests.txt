# =============================================================================
#
# База: например http://localhost:8001 (см. appointments-api/README.md)
#
# GET /health — проверка сервиса
#
# GET /slots
#   Query (опционально):
#     date — строка YYYY-MM-DD
#     service_type — одно из: consultation | demo | installation
#   Ответ: массив слотов с полями id, date, time, service_type, is_available
#
# POST /book
#   JSON body:
#     service_type — строка, должна совпадать с service_type выбранного слота
#     slot_id — int, id слота из GET /slots
#     customer_name — ФИО
#     customer_phone — телефон
#     customer_email — email
#
# Перед вызовом POST /book нужно получить актуальный slot_id из GET /slots.
# =============================================================================
